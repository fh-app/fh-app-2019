keytool -genkey -v -keystore fh-burgenland-app.keystore -alias fh-burgenland-app -keyalg RSA -keysize 2048 -validity 10000

https://www.construct.net/en/tutorials/cordova-export-signed-apk-1420/signing-release-apk-4

jarsigner -verbose -sigalg SHA1withRSA -digestalg SHA1 -keystore fh-burgenland-app.keystore app-release-unsigned.apk fh-burgenland-app

zipalign -v 4 app-release-unsigned.apk fh-burgenland-app-signed.apk

COMMANDS IN REIHENFOLGE:

phonegap build android -release

jarsigner -verbose -sigalg SHA1withRSA -digestalg SHA1 -keystore fh-burgenland-app.keystore app-release-unsigned.apk fh-burgenland-app
